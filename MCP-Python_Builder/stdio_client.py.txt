"""
Minimal-but-complete MCP client over stdio.

Connects to stdio_server.py, implements the sampling callback (so the
server's `summarize` tool has an LLM to call), and shows how to receive
progress updates from `process_items`.

Requires: pip install mcp anthropic  (and ANTHROPIC_API_KEY in the environment)
"""

import asyncio

from anthropic import AsyncAnthropic
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client
from mcp.shared.context import RequestContext
from mcp.types import CreateMessageRequestParams, CreateMessageResult, TextContent

anthropic_client = AsyncAnthropic()
MODEL = "claude-sonnet-4-6"  # the client picks (and pays for) the model, not the server

server_params = StdioServerParameters(command="uv", args=["run", "stdio_server.py"])


async def sampling_callback(
    context: RequestContext, params: CreateMessageRequestParams
) -> CreateMessageResult:
    """
    Called whenever the server asks the client to run an LLM completion.

    Note the casing: `params` uses the wire-format camelCase attributes
    (systemPrompt, maxTokens) — different from the snake_case kwargs the
    server used to build the request (system_prompt, max_tokens).
    """
    messages = [
        {"role": m.role, "content": m.content.text}
        for m in params.messages
        if m.content.type == "text"
    ]

    response = await anthropic_client.messages.create(
        model=MODEL,
        system=params.systemPrompt or "You are a helpful assistant.",
        messages=messages,
        max_tokens=params.maxTokens or 4000,
    )

    text = "".join(block.text for block in response.content if block.type == "text")
    return CreateMessageResult(
        role="assistant",
        model=MODEL,
        content=TextContent(type="text", text=text),
    )


async def on_progress(progress: float, total: float | None, message: str | None) -> None:
    pct = f"{progress}/{total}" if total else str(progress)
    print(f"[progress] {pct} {message or ''}")


async def main() -> None:
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(
            read, write, sampling_callback=sampling_callback
        ) as session:
            await session.initialize()

            summary = await session.call_tool(
                "summarize", arguments={"text_to_summarize": "lots of text"}
            )
            print(summary.content)

            result = await session.call_tool(
                "process_items",
                arguments={"items": ["a", "b", "c"]},
                progress_callback=on_progress,
            )
            print(result.content)


if __name__ == "__main__":
    asyncio.run(main())
